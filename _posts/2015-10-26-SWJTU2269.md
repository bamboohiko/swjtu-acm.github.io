由于`n <= 1e9`,暴力显然是会TLE的，但对于模算数，有这样的性质成立：

> (a*b) mod n = (a mod n)*(b mod n)

因此当`n >= 2014`时，结果都为`0`,只需计算`n < 2014`的情况即可。

<code><pre>
#include "cstdio"
#include "iostream"
#include "algorithm"

#include "cstdlib"

const int mod = 2017;
using namespace std;
int main() {
    int n;
    while (scanf("%d",&n) != EOF && n) {
        int ans = 1;
        if (n >= mod) printf("0\n");
        else {
            for (int i = 1;i <= n; i++) {
                ans = ans*i%mod;
            }
            printf("%d\n",ans);
        }
    }
	return 0;
}
